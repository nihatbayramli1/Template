    name: "default",
    pattern: "{controller=Home}/{action=Index}/{id?}");

MƏRHƏLƏ: ChefController.cs (CRUD) YARATMAQ VƏ YAZMAQSolution Explorer-də Controllers qovluğunun üzərinə siçanın sağ düyməsini vur.Add $\rightarrow$ Controller düyməsini seç.Açılan pəncərədə MVC Controller - Empty seç və aşağıdakı Add düyməsinə bas.Ad hissəsinə tam olaraq ChefController.cs yaz və Add bas.Yeni açılan bu faylın içindəki hər şeyi sil və bu təmiz CRUD kodlarını yapışdır:C#using Microsoft.AspNetCore.Authorization;
using Microsoft.AspNetCore.Mvc;
using Microsoft.EntityFrameworkCore;
using EatryCafe.Data;
using EatryCafe.Models;

namespace EatryCafe.Controllers
{
    [Authorize]
    public class ChefController : Controller
    {
        private readonly AppDbContext _context;

        public ChefController(AppDbContext context)
        {
            _context = context;
        }

        public async Task<IActionResult> Index()
        {
            var chefs = await _context.Chefs.Where(x => !x.IsDeleted).ToListAsync();
            return View(chefs);
        }

        public IActionResult Create()
        {
            return View();
        }

        [HttpPost]
        public async Task<IActionResult> Create(Chef model, IFormFile? ImageFile)
        {
            if (!ModelState.IsValid)
            {
                return View(model);
            }

            if (ImageFile != null)
            {
                string folder = Path.Combine(Directory.GetCurrentDirectory(), "wwwroot", "images");
                string fileName = Guid.NewGuid().ToString() + "_" + ImageFile.FileName;
                string filePath = Path.Combine(folder, fileName);

                using (var stream = new FileStream(filePath, FileMode.Create))
                {
                    await ImageFile.CopyToAsync(stream);
                }

                model.Image = "images/" + fileName;
            }

            await _context.Chefs.AddAsync(model);
            await _context.SaveChangesAsync();
            return RedirectToAction(nameof(Index));
        }

        public async Task<IActionResult> Delete(int id)
        {
            var chef = await _context.Chefs.FindAsync(id);
            if (chef == null) return NotFound();

            chef.IsDeleted = true;
            await _context.SaveChangesAsync();
            return RedirectToAction(nameof(Index));
        }
    }
}

MƏRHƏLƏ: AccountController.cs (LOGIN-REGISTER) YARATMAQControllers qovluğuna sağ düymə vur $\rightarrow$ Add $\rightarrow$ Controller seç.MVC Controller - Empty seçib Add bas.Adını tam olaraq AccountController.cs qoy və Add düyməsinə bas.İçindəki hər şeyi silib bu hazır loqin-qeydiyyat mexanizmini yapışdır:C#using Microsoft.AspNetCore.Identity;
using Microsoft.AspNetCore.Mvc;

namespace EatryCafe.Controllers
{
    public class AccountController : Controller
    {
        private readonly UserManager<IdentityUser> _userManager;
        private readonly SignInManager<IdentityUser> _signInManager;

        public AccountController(UserManager<IdentityUser> userManager, SignInManager<IdentityUser> signInManager)
        {
            _userManager = userManager;
            _signInManager = signInManager;
        }

        public IActionResult Register() => View();

        [HttpPost]
        public async Task<IActionResult> Register(string username, string password)
        {
            if (string.IsNullOrEmpty(username) || string.IsNullOrEmpty(password)) return View();

            var user = new IdentityUser { UserName = username };
            var result = await _userManager.CreateAsync(user, password);

            if (result.Succeeded)
            {
                await _signInManager.SignInAsync(user, isPersistent: false);
                return RedirectToAction("Index", "Home");
            }
            return View();
        }

        public IActionResult Login() => View();

        [HttpPost]
        public async Task<IActionResult> Login(string username, string password)
        {
            if (string.IsNullOrEmpty(username) || string.IsNullOrEmpty(password)) return View();

            var result = await _signInManager.PasswordSignInAsync(username, password, false, false);
            if (result.Succeeded)
            {
                return RedirectToAction("Index", "Chef");
            }
            return View();
        }

        public async Task<IActionResult> Logout()
        {
            await _signInManager.SignOutAsync();
            return RedirectToAction("Index", "Home");
        }
    }
}
